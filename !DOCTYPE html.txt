    <script>
        // 游戏状态
        const gameState = {
            diskCount: 5,
            pegs: [[], [], []],
            moves: 0,
            isAutoSolving: false,
            selectedPegIndex: null
        };

        // DOM 元素
        const diskCountInput = document.getElementById('diskCount');
        const moveCountSpan = document.getElementById('moveCount');
        const resetBtn = document.getElementById('resetBtn');
        const autoSolveBtn = document.getElementById('autoSolveBtn');
        const messageArea = document.getElementById('messageArea');
        const pegs = document.querySelectorAll('.peg');

        // 圆盘颜色数组
        const diskColors = ['#FF6B6B', '#FFA07A', '#FFD700', '#98FB98', '#87CEFA', '#DDA0DD', '#F0E68C', '#9370DB'];

        // 初始化游戏
        function initGame() {
            gameState.diskCount = parseInt(diskCountInput.value);
            gameState.pegs = [[], [], []];
            gameState.moves = 0;
            gameState.isAutoSolving = false;
            gameState.selectedPegIndex = null;

            // 初始化圆盘，大的在下，小的在上
            for (let i = gameState.diskCount; i >= 1; i--) {
                gameState.pegs[0].push(i);
            }

            updateMoveCount();
            clearMessage();
            renderBoard();
            enableControls();
        }

        // 渲染游戏板
        function renderBoard() {
            pegs.forEach((peg, index) => {
                peg.innerHTML = '';
                const disks = gameState.pegs[index];

                disks.forEach(diskSize => {
                    const disk = document.createElement('div');
                    disk.className = 'disk';
                    disk.dataset.size = diskSize;
                    // 宽度按比例计算：最小60px，最大180px
                    const width = 60 + (diskSize - 1) * (120 / (gameState.diskCount - 1));
                    disk.style.width = `${width}px`;
                    disk.style.backgroundColor = diskColors[diskSize - 1];
                    disk.addEventListener('click', () => handleDiskClick(index, diskSize));
                    peg.appendChild(disk);
                });
            });
        }

        // 更新移动次数
        function updateMoveCount() {
            moveCountSpan.textContent = gameState.moves;
        }

        // 显示消息
        function showMessage(text, type = 'warning') {
            messageArea.textContent = text;
            messageArea.className = type;
        }

        // 清除消息
        function clearMessage() {
            messageArea.textContent = '';
            messageArea.className = '';
        }

        // 处理圆盘点击
        function handleDiskClick(pegIndex, diskSize) {
            if (gameState.isAutoSolving) return;

            // 只能点击柱子最顶部的圆盘
            const topDisk = gameState.pegs[pegIndex][gameState.pegs[pegIndex].length - 1];
            if (diskSize !== topDisk) return;

            // 如果已经选中了一个柱子
            if (gameState.selectedPegIndex !== null) {
                // 点击的是同一个柱子，取消选择
                if (gameState.selectedPegIndex === pegIndex) {
                    gameState.selectedPegIndex = null;
                    pegs[pegIndex].classList.remove('selected');
                } else {
                    // 尝试移动
                    attemptMove(gameState.selectedPegIndex, pegIndex);
                    gameState.selectedPegIndex = null;
                    pegs.forEach(p => p.classList.remove('selected'));
                }
            } else {
                // 选中当前柱子
                gameState.selectedPegIndex = pegIndex;
                pegs[pegIndex].classList.add('selected');
            }
        }

        // 处理柱子点击（直接放置）
        pegs.forEach((peg, index) => {
            peg.addEventListener('click', (e) => {
                if (e.target.classList.contains('disk')) return;
                if (gameState.isAutoSolving || gameState.selectedPegIndex === null) return;

                attemptMove(gameState.selectedPegIndex, index);
                gameState.selectedPegIndex = null;
                pegs.forEach(p => p.classList.remove('selected'));
            });
        });

        // 尝试移动圆盘
        function attemptMove(fromIndex, toIndex) {
            const fromPeg = gameState.pegs[fromIndex];
            const toPeg = gameState.pegs[toIndex];
            const diskToMove = fromPeg[fromPeg.length - 1];

            // 检查移动是否合法
            if (toPeg.length === 0 || diskToMove < toPeg[toPeg.length - 1]) {
                // 执行移动
                fromPeg.pop();
                toPeg.push(diskToMove);
                gameState.moves++;
                updateMoveCount();
                clearMessage();
                renderBoard();

                // 检查是否胜利
                if (checkWin()) {
                    showVictory();
                }
            } else {
                // 非法移动
                showMessage('⚠️ Cannot place larger disk on smaller one!', 'warning');
                const disks = pegs[fromIndex].querySelectorAll('.disk');
                const topDisk = disks[disks.length - 1];
                if (topDisk) {
                    topDisk.classList.add('warning');
                    setTimeout(() => topDisk.classList.remove('warning'), 500);
                }
            }
        }

        // 检查是否胜利
        function checkWin() {
            return gameState.pegs[2].length === gameState.diskCount;
        }

        // 显示胜利消息
        function showVictory() {
            showMessage(`🎉 Victory! You solved it in ${gameState.moves} moves!`, 'victory');
            document.querySelector('.board').style.animation = 'celebrate 0.5s ease 3';
        }

        // 禁用/启用控件
        function disableControls() {
            gameState.isAutoSolving = true;
            diskCountInput.disabled = true;
            resetBtn.disabled = true;
            autoSolveBtn.disabled = true;
            pegs.forEach(peg => peg.style.pointerEvents = 'none');
        }

        function enableControls() {
            gameState.isAutoSolving = false;
            diskCountInput.disabled = false;
            resetBtn.disabled = false;
            autoSolveBtn.disabled = false;
            pegs.forEach(peg => peg.style.pointerEvents = 'auto');
        }

        // 自动求解（递归算法 + 延迟）
        async function autoSolve() {
            disableControls();
            clearMessage();
            showMessage('🔄 Auto-solving...', '');

            // 汉诺塔递归解法
            async function moveDisks(n, from, to, aux) {
                if (n === 0) return;

                await moveDisks(n - 1, from, aux, to);
                await new Promise(resolve => setTimeout(resolve, 600));
                attemptMove(from, to);
                await moveDisks(n - 1, aux, to, from);
            }

            await moveDisks(gameState.diskCount, 0, 2, 1);

            if (!checkWin()) {
                showMessage('Auto-solve complete!', 'victory');
            }
            enableControls();
        }

        // 事件绑定
        diskCountInput.addEventListener('change', () => {
            const value = parseInt(diskCountInput.value);
            if (value >= 3 && value <= 8) {
                initGame();
            } else {
                diskCountInput.value = gameState.diskCount;
            }
        });

        resetBtn.addEventListener('click', initGame);
        autoSolveBtn.addEventListener('click', autoSolve);

        // 初始化游戏
        initGame();
    </script>
